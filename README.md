// Imports the React library itself, along with the 'lazy' function using object destructuring.
// - 'React' is needed for JSX (like the <Login/> elements below).
// - 'lazy' is a built-in React utility used for dynamic importing, enabling code splitting.
import React, { lazy } from 'react';

// Defines the Login component using React's 'lazy' function.
// This implements code splitting: the actual file for the Login component 
// ('../../views/auth/Login') will not be loaded until a user navigates to the associated route.
const Login = lazy(() => import('../../views/auth/Login'));

// Defines the Register component similarly using 'lazy'.
// This ensures that the Register component file is also dynamically loaded on demand,
// improving the application's initial load time.
const Register = lazy(() => import('../../views/auth/Register'));


// Defines a constant array named 'publicRoutes'.
// This array serves as a configuration list for all navigation paths that are publicly accessible
// (i.e., do not require a user to be logged in).
const publicRoutes = [
    {
        // Start of the configuration object for the Login route.

        // 'path' specifies the URL route segment (e.g., '/login') that the router should match.
        path: '/login',

        // 'element' specifies the React component to be rendered when the 'path' matches.
        // Since 'Login' was defined using 'lazy', it must be rendered as a React element (<Login />).
        element: <Login/>,

    },
    {
        // Start of the configuration object for the Register route.

        // 'path' specifies the URL route segment for the registration page.
        path: '/register',

        // 'element' specifies the React component to be rendered.
        // It uses the lazy-loaded 'Register' component as a React element.
        element: <Register/>,
    }
]

// Exports the entire 'publicRoutes' array as the default export.
// This allows the main 'Router' component (imported in App.js) to consume this list
// and configure the routing logic based on these definitions.
export default publicRoutes;
